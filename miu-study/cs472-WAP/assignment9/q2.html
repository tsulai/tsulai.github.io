    const person = {
        name : '',
        dateOfBirth: null,
        getName: function(){
            return this.name;
        },
        setName: function(n){
            this.name = n;
        }
    }
    const employee = {
        salary: 0,
        hireDate: '',
        doJob: function(title){
            console.log(this.name + ' is a ' + title + ' who earns $' + this.salary.toFixed(2));
        }
    }
    employee.prototype = person;

    const anna = Object.create(employee,{
        name: { value: 'Anna'},
